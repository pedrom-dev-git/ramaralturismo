% Uso: \cartaobotao{url}{icone}{titulo}{subtitulo}
% Usa a key nativa "hyperurl" do tcolorbox (skins library) — gera a anotação
% /URI automaticamente, sem os problemas de \pdfendlink em modo vertical.
% Uso: \cartaobotao{url}{icone_tex}{titulo}{subtitulo}
% O 2º arg já pode vir colorido (\textcolor{...}{\faX}).
\newcommand{\cartaobotao}[4]{%
  \begin{tcolorbox}[
    enhanced,
    colback=white,
    colframe=corPrim,
    boxrule=1.2pt,
    arc=5mm,
    height=3.4cm,
    width=\linewidth,
    left=4mm, right=4mm, top=3mm, bottom=3mm,
    drop shadow={black!25,xshift=0.3mm,yshift=-0.5mm},
    halign=center, valign=center,
    hyperurl={#1},
  ]
    {\fontsize{34}{34}\selectfont #2}\\[2mm]
    {\bfseries\large\textcolor{corDk}{#3}}\\
    {\small\textcolor{black!65}{#4}}
  \end{tcolorbox}%
}

% Logo PIX aproximado: 4 losangos ao redor de um miolo (verde oficial #32BCAD).
\newcommand{\pixlogo}[1][0.55]{%
  \begin{tikzpicture}[scale=#1,baseline={(0,-0.1)}]
    % Losango central
    \fill[corPix] (0,0.25) -- (0.25,0) -- (0,-0.25) -- (-0.25,0) -- cycle;
    % Pétalas diagonais
    \fill[corPix] (0.55,0.55) -- (0.80,0.30) -- (0.55,0.05) -- (0.30,0.30) -- cycle;
    \fill[corPix] (-0.55,0.55) -- (-0.30,0.30) -- (-0.55,0.05) -- (-0.80,0.30) -- cycle;
    \fill[corPix] (0.55,-0.55) -- (0.30,-0.30) -- (0.55,-0.05) -- (0.80,-0.30) -- cycle;
    \fill[corPix] (-0.55,-0.55) -- (-0.80,-0.30) -- (-0.55,-0.05) -- (-0.30,-0.30) -- cycle;
  \end{tikzpicture}%
}
